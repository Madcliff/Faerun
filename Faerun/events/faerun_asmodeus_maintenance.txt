character_event = {
	id = asmodeusmaint.3
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		OR = {
			mother = { 
				OR = { 
					trait = infernal_beauty 
					trait = infernal_wit 
					trait = infernal_strength 
					trait = infernal_contractee 
				}
				dynasty = ROOT
			}
			true_father = { 
				OR = { 
                    trait = infernal_beauty 
                    trait = infernal_wit 
					trait = infernal_strength 
					trait = infernal_contractee 
				}
                dynasty = ROOT
			}
		}
		NOT = { trait = bastard }
	}

	immediate = { 
		if = {
			limit = {
				mother = { 
                    trait = infernal_beauty
                    dynasty = ROOT  
                }
			}
			add_trait = infernal_beauty
			add_trait = beautiful
			break = yes
		}
		else_if = {
			limit = {
				mother = { 
                    trait = infernal_wit
                    dynasty = ROOT  
                }
			}
			add_trait = infernal_wit
			add_trait = genius
			break = yes
		}
		else_if = {
			limit = {
				mother = { 
                    trait = infernal_strength
                    dynasty = ROOT  
                }
			}
			add_trait = infernal_strength
			add_trait = strong
			break = yes
		}		
		else_if = {
			limit = {
				mother = { 
                    trait = infernal_contractee
                    dynasty = ROOT  
                }
			}
			add_trait = infernal_contractee
			break = yes
		}
		else_if = {
			limit = {
				true_father = { 
                    trait = infernal_beauty
                    dynasty = ROOT  
                }
			}
			add_trait = infernal_beauty 
			add_trait = beautiful
			break = yes
		}
		else_if = {
			limit = {
				true_father = { 
                    trait = infernal_wit
                    dynasty = ROOT  
                }
			}
			add_trait = infernal_wit
			add_trait = genius
			break = yes
		}
		else_if = {
			limit = {
				true_father = { 
                    trait = infernal_strength
                    dynasty = ROOT  
                }
			}
			add_trait = infernal_strength
			add_trait = strong
			break = yes
		}
		else_if = {
			limit = {
				true_father = { 
                    trait = infernal_contractee
                    dynasty = ROOT  
                }
			}
			add_trait = infernal_contractee
			break = yes
		}
	}
}
