character_event = {
    id = MASK.100
    hide_window = yes
    is_triggered_only = yes
    trigger = {
    	FROM = { leader = { society_member_of = night_masks } }
    }
    immediate = {
    	if = { #Find a non-prisoner member
    		limit = { FROM = { leader = { prisoner = yes } } }
    		FROM = {
    			any_society_member = {
    				limit = {
    					society_member_of = night_masks
    					society_rank == 4
    					prisoner = no
    				}
    				character_event = { id = MASK.101 }
    				break = yes
    			}
    			any_society_member = {
    				limit = {
    					society_member_of = night_masks
    					society_rank == 3
    					prisoner = no
    				}
    				character_event = { id = MASK.101 }
    				break = yes
    			}
    			any_society_member = {
    				limit = {
    					society_member_of = night_masks
    					society_rank == 2
    					prisoner = no
    				}
    				character_event = { id = MASK.101 }
    				break = yes
    			}
    			any_society_member = {
    				limit = {
    					society_member_of = night_masks
    					society_rank == 1

    				}

	id = MASK.110
	hide_window = yes

	is_triggered_only = yes

	immediate = {
		if = {
			limit = { #if he's already a member, will force a rank up 25% of the time
				ai = yes
				prisoner = no
				NOT = { trait = incapable }
				society_member_of = night_masks
				society_rank <= 3
				society_can_rank_up = yes
			}
			random_list = {
				75 = {}
				25 = {
					society_rank_up = 1
				}
			}
		}
		if = { #force the joining event on AI
			limit = {
				ai = yes
				prisoner = no
				NOT = { trait = incapable }
				is_adult = yes
				is_in_society = no
				controls_religion = no
				block_general_event_trigger = no
			}
			character_event = { id = MASK.111 }
		}
	}
}
