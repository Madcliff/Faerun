harass_tactic = {	# exclusive for light cavalry
	days = 6
	sprite = 4
	group = harass

	trigger = {
		phase = skirmish
		OR = {
			troops = {
				who = light_cavalry
				value = 0.01
			}
			troops = {
				who = camel_cavalry
				value = 0.01
			}
			troops = {
				who = beast_troops
				value = 0.01
			}
		}
		OR = {
			NOT = {
				OR = {
					troops = {
						who = light_cavalry
						value = 0.45
					}
					troops = {
						who = camel_cavalry
						value = 0.45
					}
					troops = {
						who = beast_troops
						value = 0.45
					}
				}
			}
			AND = {
				flank_has_leader = yes
				leader = { martial = 12 }

						troops = {
							who = light_cavalry
							value = 0.75

							who = camel_cavalry

						}
						troops = {
							who = beast_troops
							value = 0.75
