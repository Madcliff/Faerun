spell_resistance_check = {
	random_list = {
		1 = {
			modifier = {
				factor = 2
				attribute_diff = {
					character = FROM
					attribute = learning
					value >= 5
				}
			}
			modifier = { # Stacking with previous is on purpose
				factor = 3
				attribute_diff = {
					character = FROM
					attribute = learning
					value >= 10
				}
			}
			modifier = {
				factor = 4
				attribute_diff = {
					character = FROM
					attribute = learning
					value >= 15
				}
			}
			modifier = {
				factor = 5
				attribute_diff = {
					character = FROM
					attribute = learning
					value >= 20
				}
			}
			modifier = {
				factor = 4
				OR = {
					has_character_modifier = spell_mantle_mod
					any_artifact = {
						artifact_type = mantle_of_spell_resistance
						is_artifact_equipped = yes
					}
					any_artifact = {
						artifact_type = robe_of_archmagi
						is_artifact_equipped = yes
					}
					any_artifact = {
						artifact_type = spellguard_shield
						is_artifact_equipped = yes
					}
					any_artifact = {
						artifact_type = holy_avenger
						is_artifact_equipped = yes
					}
					any_artifact = {
						artifact_type = githyanki_silver_sword
						is_artifact_equipped = yes
					}
					any_artifact = {
						artifact_type = luck_blade
						is_artifact_equipped = yes
					}
					any_artifact = {
						artifact_type = staff_of_the_magi
						is_artifact_equipped = yes
					}
				}
			}
			#This just gives the target a bad opinion, tells the caster it failed and hopefuly aborts the effects of the decision
			opinion = {
				modifier = cast_spell_on_me
				who = FROM
				years = 1
			}
			FROM = {
				character_event = { id = spellaux.1 }
			}
			break = yes
		}
		1 = {
			modifier = {
				factor = 2
				FROM = {

						character = ROOT

						value >= 5

						value >= 10

			}

				FROM = {
					attribute_diff = {
						character = ROOT

						value >= 15
					}
				}
			}
